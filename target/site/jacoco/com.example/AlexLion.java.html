package com.example;

import java.util.Arrays;
import java.util.List;

public class AlexLion extends Lion {
    // Друзья Алекса
    private static final List<String> FRIENDS = Arrays.asList("Марти", "Глория", "Мелман");
    private static final String PLACE_OF_LIVING = "Нью-Йоркский зоопарк";

    // Конструктор, передаем фиксированные значения для Алекса
    public AlexLion(Feline feline) {
        // Вызываем конструктор родителя с параметрами "самец" и переданный Feline
        super("самец", feline);
    }

    // Метод для получения списка друзей
    public List<String> getFriends() {
        return FRIENDS;
    }

    // Метод для получения места проживания
    public String getPlaceOfLiving() {
        return PLACE_OF_LIVING;
    }

    // Переопределяем метод getKittens(), так как у Алекса нет котят
    @Override
    public int getKittens() {
        return 0; // У Алекса нет львят
    }
}
